Для интеграции встройте в требуемые страницы код (С API можно ознакомиться внутри скрипта): 
                  window.addEventListener("load", () => {
                    var script = document.createElement("script");
                    script.setAttribute("data-stream-helper-secret", 'Ваш_уникальный_ключ');
                    window.streamHelper_customHandlers = {}; // Пользовательские обработчики
                    script.src = "https://klybok.net:3001/js/remote.js";
                    document.body.appendChild(script);
                  });
